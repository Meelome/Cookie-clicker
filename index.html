// Создаем элемент с изображением капибары const capybaraImage = document.createElement('img'); capybaraImage.src = 'https://i.imgur.com/2w0r3oA.jpg'; document.body.appendChild(capybaraImage); // Создаем элемент для отображения счетчика кликов const clickCounter = document.createElement('p'); clickCounter.textContent = 'Количество кликов: 0'; document.body.appendChild(clickCounter); let clicks = 0; let canClick = true; // Функция для обработки кликов function handleClick() { if (canClick) { clicks++; clickCounter.textContent = `Количество кликов: ${clicks}`; canClick = false; setTimeout(() => { canClick = true; }, 30000); // Задержка в 30 секунд } } // Добавляем обработчик кликов на изображение капибары capybaraImage.addEventListener('click', handleClick);
